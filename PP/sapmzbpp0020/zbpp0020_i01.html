MODULE USER_COMMAND_0100 INPUT.
  CASE OK_CODE.
    WHEN 'BACK' OR 'EXIT' OR 'CANCEL'. " BACK, EXIT, CANCEL 누르는 경우 뒤로가기
      LEAVE TO SCREEN 0.

    WHEN 'CREATE'. " 생성 버튼 클릭
      CALL SCREEN 200. " 생성 화면 이동
      CLEAR: ZTBPP0070, GT_ZTBPP0071. " 사용한 변수 초기화

    WHEN 'MODIFY'. " 수정 버튼 클릭
      CLEAR: GT_EDIT_ZTBPP0071.
      PERFORM MODIF_BOM. " 수정 진행 SUB ROUTINE 실행
      CLEAR: ZTBPP0070, GT_EDIT_ZTBPP0071. " 사용한 변수 초기화

    WHEN 'DELETE'. " 삭제 버튼 클릭
      PERFORM DEL_BOM.


    WHEN 'REFRESH'. " 초기화 버튼 클릭
      PERFORM CLEAR_ALL.

    WHEN 'BTN'. " 조회 버튼 클릭
      PERFORM GET_DATA.

  ENDCASE.
ENDMODULE.

MODULE EXIT_CREATE INPUT.

    WHEN 'BACK' OR 'CANCEL'.
      PERFORM MESSAGE_CANCEL_C.
      LEAVE TO SCREEN 0.
    WHEN 'EXIT'.

  ENDCASE.
ENDMODULE.
