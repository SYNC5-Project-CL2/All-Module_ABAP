MODULE get_data_for_alvs OUTPUT.

  DATA: lt_gjahr TYPE RANGE OF ztbfi0030-gjahr,
        ls_gjahr LIKE LINE OF lt_gjahr,
        lt_blart TYPE RANGE OF ztbfi0030-blart,
        ls_blart LIKE LINE OF lt_blart.

  IF pa_yr is INITIAL.

  ELSE.

  ls_gjahr-low = pa_yr.
  ls_gjahr-sign = 'I'.
  ls_gjahr-option = 'EQ'.
  APPEND ls_gjahr to lt_gjahr.

  ENDIF.

  IF pa_belty is INITIAL.

  ELSE.

  ls_blart-low = pa_belty.
  ls_blart-sign = 'I'.
  ls_blart-option = 'EQ'.
  APPEND ls_blart to lt_blart.

  ENDIF.



  SELECT *
    FROM ztbfi0030
    into CORRESPONDING FIELDS OF TABLE gt_alv1
    WHERE COMCODE = pa_bpc
      and gjahr in lt_gjahr
      and budat in so_dat
      and blart in lt_blart
      and belnr in so_belnr.

    sort gt_alv1 by belnr ASCENDING.

ENDMODULE.
