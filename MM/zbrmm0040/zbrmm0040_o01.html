MODULE GET_BPINFO OUTPUT.
  SELECT SINGLE *
    FROM ZTBSD1050 AS A
    JOIN ZTBSD1051 AS B
      ON A~BPCODE EQ B~BPCODE
    JOIN ZTBFI0040 AS C
      ON A~ZTERM EQ C~ZTERM
    JOIN ZTBFI0060 AS D
      ON A~BANKCODE EQ D~BANKCODE
   WHERE A~BPCODE EQ @PA_BPCO
    INTO CORRESPONDING FIELDS OF @ZSBMM0060.

ENDMODULE.
