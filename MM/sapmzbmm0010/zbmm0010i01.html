MODULE USER_COMMAND_0100 INPUT.
  CASE OK_CODE.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

      " '조회' 버튼 클릭 시.
    WHEN 'BTN1'.
      PERFORM GET_DATA.

      " '생성' 버튼 클릭 시.
    WHEN 'CREATE'.
      CALL SCREEN 110
        STARTING AT 80 7.

      " '수정' 버튼 클릭 시.
    WHEN 'CHANGE' OR 'DELETE'.
      PERFORM DATA_SELECTED_ROW.
      CLEAR GS_DATA100.

      " '새로고침' 버튼 클릭 시.
    WHEN 'REFRESH'.
      PERFORM SET_REFRESH.
  ENDCASE.
ENDMODULE.
