MODULE USER_COMMAND_0100 INPUT.
  CASE OK_CODE.
    WHEN 'BACK' OR 'CANCEL' OR 'EXIT'.
      LEAVE TO SCREEN 0.

    WHEN 'CREATE'. " 생성 버튼 클릭 시
      CALL SCREEN 110
        STARTING AT 5 5.

    WHEN 'MODIFY' OR 'DELETE'. " 수정 / 삭제 버튼 클릭 시
      PERFORM ZBMM_GET_SEL_ROW.

    WHEN 'BTN'.  " 조회 버튼 클릭 시
      PERFORM ZBMM_GET_STRDATA. " 조회 서브루틴 실행

    WHEN 'REFRESH'.
      PERFORM ZBMM_CLEAR_ALL.

  ENDCASE.
ENDMODULE.
